1.function declaration

2.function expression

3.arrow function

        1.function fun_name(parameters){
            //body of the function
        }
        fun_name(arguments);

        2.const fun_name = function(parameters){
            //body of function
        }
        fun_name(arguments);

        3.const fun_name = (parameters) => {
            //body of function
        }
        fun_name(arguments);
